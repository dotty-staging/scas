val mml = scas.application.MathML("mmlscala.xsl")
val manager = new javax.script.ScriptEngineManager
val engine = manager.getEngineByName("scas")
Map("x+x" -> "<math><apply><times/><cn>2</cn><ci>x</ci></apply></math>",
"1" -> "1",
"1+x" -> "<math><apply><plus/><cn>1</cn><ci>x</ci></apply></math>",
"1/4" -> "<math><cn type="rational">1<sep/>4</cn></math>") map { r =>
  val (a, b) = r
  assert(mml(engine.eval(a)) == b)
}
